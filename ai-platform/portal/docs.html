Official SDKs

Install the Eior SDK for your language — no base_url config needed, works out of the box.

Step 1 — Install

pip install eior

Step 2 — Get your API key from the dashboard

Step 3 — Use it

from eior import Eior

client = Eior(api_key="sk-your-key")

# Chat
response = client.chat.completions.create(
    model="eior-chat",
    messages=[{"role": "user", "content": "Hello!"}],
)
print(response.choices[0].message.content)

# Streaming
stream = client.chat.completions.create(
    model="eior-chat",
    messages=[{"role": "user", "content": "Count to 10"}],
    stream=True,
)
for chunk in stream:
    print(chunk.choices[0].delta.content or "", end="", flush=True)

# Async
from eior import AsyncEior
import asyncio

async def main():
    client = AsyncEior(api_key="sk-your-key")
    res = await client.chat.completions.create(
        model="eior-chat",
        messages=[{"role": "user", "content": "Hello!"}],
    )
    print(res.choices[0].message.content)

asyncio.run(main())

# Vision (photo analysis — e.g. outfit checker)
response = client.chat.completions.create(
    model="eior-vision",
    messages=[{
        "role": "user",
        "content": [
            {"type": "text", "text": "What outfit would suit this person?"},
            {"type": "image_url", "image_url": {"url": "https://example.com/photo.jpg"}}
        ]
    }],
)
print(response.choices[0].message.content)

Authentication

All generation endpoints require an API key in the Authorization header.

Authorization: Bearer sk-your-api-key

Chat Completions

POST/v1/chat/completions

Generate a chat response using Eior's AI model. OpenAI-compatible format.

{
  "model": "eior-chat",       // optional
  "messages": [
    { "role": "system",    "content": "You are a helpful assistant." },
    { "role": "user",      "content": "What is the capital of France?" }
  ]
}

Example:

curl https://api-production-2b12.up.railway.app/v1/chat/completions \
  -H "Authorization: Bearer sk-your-key" \
  -H "Content-Type: application/json" \
  -d '{
    "messages": [{"role":"user","content":"Say hello in French"}]
  }'

Response:

{
  "id": "chatcmpl-1234567890",
  "object": "chat.completion",
  "model": "eior-chat",
  "choices": [{
    "message": { "role": "assistant", "content": "Bonjour!" },
    "finish_reason": "stop"
  }],
  "usage": { "prompt_tokens": 5, "completion_tokens": 3, "total_tokens": 8 }
}

Function Calling & Tool Use

Attach functions as tools so the model can call them. Works exactly like OpenAI's tool-use format.

{
  "model": "eior-chat",
  "messages": [{"role":"user","content":"What is the weather in London?"}],
  "tools": [{
    "type": "function",
    "function": {
      "name": "get_weather",
      "description": "Get the current weather for a city",
      "parameters": {
        "type": "object",
        "properties": {
          "location": { "type": "string", "description": "City name" },
          "unit":     { "type": "string", "enum": ["celsius","fahrenheit"] }
        },
        "required": ["location"]
      }
    }
  }],
  "tool_choice": "auto"
}

Response when tool is called:

{
  "choices": [{
    "finish_reason": "tool_calls",
    "message": {
      "role": "assistant",
      "content": null,
      "tool_calls": [{
        "id": "call_abc123",
        "type": "function",
        "function": { "name": "get_weather", "arguments": "{\"location\":\"London\"}" }
      }]
    }
  }]
}

Python with tool use:

from eior import Eior

client = Eior(api_key="sk-your-key")

response = client.chat.completions.create(
    model="eior-chat",
    messages=[{"role": "user", "content": "What's the weather in Paris?"}],
    tools=[{
        "type": "function",
        "function": {
            "name": "get_weather",
            "description": "Get weather for a city",
            "parameters": {
                "type": "object",
                "properties": { "location": {"type": "string"} },
                "required": ["location"]
            }
        }
    }],
    tool_choice="auto",
)

if response.choices[0].finish_reason == "tool_calls":
    tool_call = response.choices[0].message.tool_calls[0]
    print(f"Model wants to call: {tool_call.function.name}")
    print(f"With args: {tool_call.function.arguments}")

Embeddings

POST/v1/embeddings

Generate vector embeddings for text — for semantic search, RAG, and similarity tasks.

{
  "input": "The quick brown fox jumps",  // string or array of strings
  "model": "eior-embed"                 // optional
}
curl https://api-production-2b12.up.railway.app/v1/embeddings \
  -H "Authorization: Bearer sk-your-key" \
  -H "Content-Type: application/json" \
  -d '{"input":["Hello world","How are you"],"model":"eior-embed"}'

Response:

{
  "object": "list",
  "model": "eior-embed",
  "data": [
    { "object": "embedding", "index": 0, "embedding": [0.023, -0.051, ...] },
    { "object": "embedding", "index": 1, "embedding": [0.018,  0.042, ...] }
  ],
  "usage": { "prompt_tokens": 6, "total_tokens": 6 }
}

Python:

from eior import Eior

client = Eior(api_key="sk-your-key")

result = client.embeddings.create(
    model="eior-embed",
    input=["Semantic search query", "Document text here"],
)

for item in result.data:
    print(f"Vector length: {len(item.embedding)}")

Audio

POST/v1/audio/speech  Text-to-Speech

Convert text to natural-sounding speech. Returns an audio file (MP3).

{
  "input": "Hello! Welcome to Eior, your AI platform.",
  "voice": "alloy",   // alloy | echo | fable | onyx | nova | shimmer
  "model": "eior-tts" // optional
}
curl https://api-production-2b12.up.railway.app/v1/audio/speech \
  -H "Authorization: Bearer sk-your-key" \
  -H "Content-Type: application/json" \
  -d '{"input":"Hello world","voice":"nova"}' \
  --output speech.mp3

Python:

from eior import Eior

client = Eior(api_key="sk-your-key")

with client.audio.speech.with_streaming_response.create(
    model="eior-tts",
    voice="nova",
    input="The rain in Spain stays mainly in the plain.",
) as res:
    res.stream_to_file("output.mp3")

POST/v1/audio/transcriptions  Speech-to-Text (Whisper)

Transcribe audio to text using Whisper. Accepts multipart/form-data file upload.

curl https://api-production-2b12.up.railway.app/v1/audio/transcriptions \
  -H "Authorization: Bearer sk-your-key" \
  -F "file=@audio.mp3" \
  -F "model=whisper-1"

Python:

from eior import Eior

client = Eior(api_key="sk-your-key")

with open("audio.mp3", "rb") as f:
    result = client.audio.transcriptions.create(
        model="whisper-1",
        file=f,
    )
print(result.text)

Assistants API

Build powerful AI agents with persistent memory and tool use. Create an assistant, attach it to a thread, run it.

Assistants

POST /v1/assistants GET /v1/assistants GET /v1/assistants/:id POST /v1/assistants/:id  (update) DELETE /v1/assistants/:id

Threads & Messages

POST /v1/threads GET /v1/threads/:id DELETE /v1/threads/:id POST /v1/threads/:id/messages GET /v1/threads/:id/messages GET /v1/threads/:id/messages/:msg_id

Runs

POST /v1/threads/:id/runs GET /v1/threads/:id/runs GET /v1/threads/:id/runs/:run_id POST /v1/threads/:id/runs/:run_id/cancel

Full example (Python):

from eior import Eior

client = Eior(api_key="sk-your-key")

# 1. Create an assistant
assistant = client.beta.assistants.create(
    name="Eior Helper",
    instructions="You are a helpful AI assistant.",
    model="eior-chat",
)

# 2. Create a thread
thread = client.beta.threads.create()

# 3. Add a message to the thread
client.beta.threads.messages.create(
    thread_id=thread.id,
    role="user",
    content="Explain quantum computing in simple terms.",
)

# 4. Run the assistant
run = client.beta.threads.runs.create_and_poll(
    thread_id=thread.id,
    assistant_id=assistant.id,
)

# 5. Get the response
if run.status == "completed":
    messages = client.beta.threads.messages.list(thread_id=thread.id)
    for msg in messages.data:
        print(f"{msg.role}: {msg.content[0].text.value}")

Error Format

All errors return a consistent JSON envelope:

{
  "error": "Human-readable message",
  "code":  "MACHINE_READABLE_CODE"
}
StatusCodeMeaning
401INVALID_API_KEYAPI key is invalid or revoked
401INVALID_TOKENFirebase ID token is invalid/expired
400VALIDATION_ERRORRequest body failed validation
429RATE_LIMIT_EXCEEDED60 req/min limit reached — check Retry-After
502AI_SERVICE_ERRORUpstream AI provider unavailable

JavaScript Example

const API_KEY = 'sk-your-key';
const BASE    = 'https://api-production-2b12.up.railway.app';

// Chat
const chat = await fetch(`${BASE}/v1/chat/completions`, {
  method:  'POST',
  headers: { Authorization: `Bearer ${API_KEY}`, 'Content-Type': 'application/json' },

}).then(r => r.json());

// Image
const img = await fetch(`${BASE}/v1/images/generations`, {
  method:  'POST',
  headers: { Authorization: `Bearer ${API_KEY}`, 'Content-Type': 'application/json' },

}).then(r => r.json());

console.log(img.data[0].url);
